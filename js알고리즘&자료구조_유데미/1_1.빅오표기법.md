## 코드 실행 시간 측정하기

```js
function uptoAdd(n){
    let total = 0;
    for(let i = 1; i < n; i++){
        total += 1;
    }
    return total;
}

let t1 = performance.now();
uptoAdd(1000000000);     // 10억까지의 수들의 합.
let t2 = performance.now();     // 밀리세컨으로 측정됨
console.log(`Time Elapsed : ${(t2 - t1) / 1000} seconds`)     //밀리세컨이므로 1000으로 나눠주면 초

```
